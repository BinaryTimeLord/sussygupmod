foreign_influence = {
	SOV_consolidate_communist_governments = {

		icon = generic_nationalism

		allowed = {
			original_tag  = SOV
		}

		available = {
			FROM = {
				any_state = {
					is_core_of = FROM
					is_controlled_by = ROOT
					compliance > 0.25
				}
			}
		}
        visible = {
			original_tag = SOV
			has_completed_focus = SOV_consolidate_communist_governments
        }
		target_non_existing = yes
		target_array = occupied_countries
		
		cost = 100
		
		fire_only_once = yes

		
		modifier = {
		}
		ai_will_do = {
			base = 10
			modifier = {
				FROM = { 
					OR = {
						original_tag = EST
						original_tag = LAT
						original_tag = LIT
						original_tag = FIN
					}
				} 
				factor = -1
			}
		}
		
		complete_effect = {
			if = {
				limit = { FROM = { original_tag = POL } }
				every_state = {
					limit = { 
						is_core_of = SOV 
						is_core_of = POL
					}
					remove_core_of = POL
				}
			}
			if = {
			# if country_to_initiate does not exist, simply release it as a puppet
				limit = { 
					var:country_to_initiate = {
						exists = no
					}
				}
				var:country_to_initiate = { set_variable = { collaboration_formed_by = PREV } }
				if = {
					limit = { 
						has_completed_focus = SOV_union_of_soviet_socialist_republics
					}
					release_autonomy = {
						target = country_to_initiate
						autonomy_state = autonomy_soviet_socialist_republic
						freedom_level = 0.5
						release_non_owned_controlled = yes
					}
				}
				else = {
					release_autonomy = {
						target = country_to_initiate
						autonomy_state = autonomy_peoples_democracy
						freedom_level = 0.5
						release_non_owned_controlled = yes
					}
				}
			}
			else = {
				# if country_to_initiate exists, create a dynamic country as our new puppet
				
				create_dynamic_country = {
					original_tag = country_to_initiate
				
					set_variable = { collaboration_formed_by = PREV }
					set_temp_variable = { new_country = this }
					
					PREV = {
						every_controlled_state = {
							limit = { occupied_country_tag = country_to_initiate }
							
							var:new_country = {
								transfer_state = PREV
							}
						}
						
						puppet = new_country
						if = {
							limit = { 
								has_completed_focus = SOV_union_of_soviet_socialist_republics
							}
							set_autonomy = {
								target = new_country
								autonomy_state = autonomy_soviet_socialist_republic
								freedom_level = 0.5
							}
						}
					}
				}
			}
		}
	}
